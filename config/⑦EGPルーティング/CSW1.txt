hostname CSW1

! ============================================================
! BGP（AS65001 / iBGPフルメッシュ）
!   自機 Lo0=10.1.254.3。ピア: DSW1(.1) DSW2(.2) CSW2(.4) R1(.5) R2(.6)

router bgp 65001
 neighbor 10.1.254.1 remote-as 65001
 neighbor 10.1.254.1 update-source Loopback0
 neighbor 10.1.254.2 remote-as 65001
 neighbor 10.1.254.2 update-source Loopback0
 neighbor 10.1.254.4 remote-as 65001
 neighbor 10.1.254.4 update-source Loopback0
 neighbor 10.1.254.5 remote-as 65001
 neighbor 10.1.254.5 update-source Loopback0
 neighbor 10.1.254.6 remote-as 65001
 neighbor 10.1.254.6 update-source Loopback0
