! ============================================================
! ③ L３項目 : DSW1 （ディストリビューションL3SW）
! 出典: 設計書_md/03_論理設計.md（論理設計②③④⑤⑥⑦）, 04_耐障害性.md
!       設問 06_HSRPクライアントGW
! 内容: SVI+IP / ループバック / DSW-CSW間L3 EtherChannel(/30) / HSRP / DHCP
! 前提: L2(VLAN/トランク/STP) は ②、OSPF は ④ で設定済みとする
! ============================================================
!
! ------------------------------------------------------------
! ■ 不明点・確認事項（実機設定時に確認）
!  - Po番号: Po1=CSW1向け, Po2=CSW2向け（③のL3）, Po3=DSW間(②のL2)。設問08準拠。
!  - HSRP track の Po1/Po2 は本ファイルで定義するCSW向けL3 EtherChannel。
!    「CSW1,2両方の接続障害時のみ切替」を priority差10・track decrement 6×2 で実現
!    （1本喪失=110-6=104>100 維持 / 2本喪失=110-12=98<100 切替）。設問06準拠。
!  - HSRPグループ番号は VLAN番号と同じにしている（VLAN10→grp10 等）。
!  - DHCPサーバは論理設計②に従い DSW1=VLAN10,20,30 のみ。DNS等の配布は設計に
!    記載がないため未設定（必要なら dns-server 等を追加）。
!  - port-channel load-balance は②で src-dst-mac をグローバル設定済み。L3リンクは
!    /30・2メンバのため実質バランスに影響は小さい。
!  - SVIの no shutdown は明示のため付与。
! ------------------------------------------------------------

hostname DSW1
!
! ============================================================
! DHCP（DSW1担当: VLAN10,20,30 / 論理設計②③）
!   配布範囲 .101-.199、既定GW=HSRP VIP .200
! ============================================================
ip dhcp excluded-address 10.1.10.1 10.1.10.100
ip dhcp excluded-address 10.1.10.200 10.1.10.254
ip dhcp excluded-address 10.1.20.1 10.1.20.100
ip dhcp excluded-address 10.1.20.200 10.1.20.254
ip dhcp excluded-address 10.1.30.1 10.1.30.100
ip dhcp excluded-address 10.1.30.200 10.1.30.254
!
ip dhcp pool VLAN10
 network 10.1.10.0 255.255.255.0
 default-router 10.1.10.200
!
ip dhcp pool VLAN20
 network 10.1.20.0 255.255.255.0
 default-router 10.1.20.200
!
ip dhcp pool VLAN30
 network 10.1.30.0 255.255.255.0
 default-router 10.1.30.200
!
! ============================================================
! ループバック（論理設計⑦: DSW1 = 10.1.254.1/32 / OSPF router-id等に使用）
! ============================================================
interface Loopback0
 ip address 10.1.254.1 255.255.255.255
!
! ============================================================
! DSW-CSW間 L3 EtherChannel（/30 / 論理設計⑥ / 設問08）
!   Po1: DSW1-CSW1  Fa1/0/1-2  10.1.255.0/30  DSW1=.2
!   Po2: DSW1-CSW2  Fa1/0/3-4  10.1.255.16/30 DSW1=.18
! ============================================================
interface range FastEthernet1/0/1 - 2
 no switchport
 channel-protocol lacp
 channel-group 1 mode active
 description To CSW1 (Po1 member)
!
interface Port-channel1
 no switchport
 ip address 10.1.255.2 255.255.255.252
 description To CSW1 L3 EtherChannel
!
interface range FastEthernet1/0/3 - 4
 no switchport
 channel-protocol lacp
 channel-group 2 mode active
 description To CSW2 (Po2 member)
!
interface Port-channel2
 no switchport
 ip address 10.1.255.18 255.255.255.252
 description To CSW2 L3 EtherChannel
!
! ============================================================
! クライアントVLAN SVI + HSRP（論理設計③④ / 設問06）
!   DSW1 = VLAN10,20,30 の Active（priority 110）
!         = VLAN40,50,60 の Standby（priority 100）
!   共通: VIP=.200, preempt, timers 200/750ms
!   Active側のみ CSW上りPo1/Po2 を track（decrement 6）
! ============================================================
interface Vlan10
 ip address 10.1.10.253 255.255.255.0
 no shutdown
 standby 10 ip 10.1.10.200
 standby 10 priority 110
 standby 10 preempt
 standby 10 timers msec 200 msec 750
 standby 10 track Port-channel1 decrement 6
 standby 10 track Port-channel2 decrement 6
!
interface Vlan20
 ip address 10.1.20.253 255.255.255.0
 no shutdown
 standby 20 ip 10.1.20.200
 standby 20 priority 110
 standby 20 preempt
 standby 20 timers msec 200 msec 750
 standby 20 track Port-channel1 decrement 6
 standby 20 track Port-channel2 decrement 6
!
interface Vlan30
 ip address 10.1.30.253 255.255.255.0
 no shutdown
 standby 30 ip 10.1.30.200
 standby 30 priority 110
 standby 30 preempt
 standby 30 timers msec 200 msec 750
 standby 30 track Port-channel1 decrement 6
 standby 30 track Port-channel2 decrement 6
!
interface Vlan40
 ip address 10.1.40.253 255.255.255.0
 no shutdown
 standby 40 ip 10.1.40.200
 standby 40 priority 100
 standby 40 preempt
 standby 40 timers msec 200 msec 750
!
interface Vlan50
 ip address 10.1.50.253 255.255.255.0
 no shutdown
 standby 50 ip 10.1.50.200
 standby 50 priority 100
 standby 50 preempt
 standby 50 timers msec 200 msec 750
!
interface Vlan60
 ip address 10.1.60.253 255.255.255.0
 no shutdown
 standby 60 ip 10.1.60.200
 standby 60 priority 100
 standby 60 preempt
 standby 60 timers msec 200 msec 750
!
! ============================================================
! 管理VLAN99 SVI + HSRP（論理設計⑤ / DSW1 Active）
!   10.1.99.0/29  DSW1=.3  VIP=.6
! ============================================================
interface Vlan99
 ip address 10.1.99.3 255.255.255.248
 no shutdown
 standby 99 ip 10.1.99.6
 standby 99 priority 110
 standby 99 preempt
 standby 99 timers msec 200 msec 750
 standby 99 track Port-channel1 decrement 6
 standby 99 track Port-channel2 decrement 6
!
end
