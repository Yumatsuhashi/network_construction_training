! ============================================================
! ⑦ EGPルーティング : R2 （拠点間接続ルータ）
! 出典: 設計書_md/08_拠点間接続設計.md
!       設問 18_R2のBGP / 19_iBGPフルメッシュ / 20_集約 / 21_LocalPref / 22_ASパスプリペンド
! 内容: eBGP(対IP-VPN) / iBGPフルメッシュ / 集約10.1.0.0/16 / 経路調整
! 前提: I/F・ループバックは③、OSPF(ループバック到達性)は④で設定済み
! ============================================================
!
! ------------------------------------------------------------
! ■ 不明点・確認事項（実機設定時に確認）
!  - 【設問に無いが必須】iBGPネイバーに next-hop-self を付与（理由はR1.txt参照）。
!    R2はLo0 10.1.254.6 をnext-hopに書き換え。
!  - 【集約方式/設問20】方式1(network + ip route Null0)を採用（理由はR1.txt参照）。
!  - R2の担当:
!    ・拠点2(10.2.0.0/16)宛をR2優先 → Local Preference 200 を in で付与（設問21）
!    ・キャリアからの戻り経路をR1優先 → R2広告にAS-PATHプリペンド out（設問22）
!  - プリペンド回数は3回（例）。確実に劣後させる調整値で絶対的正解はない（設問22）。
!  - 192.168.12.0/30 は内部BGP表に出さない（設問19条件を満たす）。
! ------------------------------------------------------------

hostname R2
!
! ============================================================
! 経路調整用 prefix-list / route-map
!   設問21: 拠点2宛をR2優先（Local Preference in）
!   設問22: 戻り経路をR1優先（AS-PATHプリペンド out）
! ============================================================
ip prefix-list SITE2-ROUTE permit 10.2.0.0/16
!
route-map SET-LOCALPREF-IN permit 10
 match ip address prefix-list SITE2-ROUTE
 set local-preference 200
route-map SET-LOCALPREF-IN permit 20
!
route-map PREPEND-OUT permit 10
 set as-path prepend 65001 65001 65001
!
! ============================================================
! 集約ルート用 静的経路（方式1 / 設問20）
! ============================================================
ip route 10.1.0.0 255.255.0.0 Null0
!
! ============================================================
! BGP（AS65001）
! ============================================================
router bgp 65001
 ! --- eBGP: IP-VPN網（対向AS64999, 認証cisco123 / 設問18）---
 neighbor 192.168.12.1 remote-as 64999
 neighbor 192.168.12.1 password cisco123
 neighbor 192.168.12.1 route-map SET-LOCALPREF-IN in
 neighbor 192.168.12.1 route-map PREPEND-OUT out
 !
 ! --- iBGP フルメッシュ（拠点内, Loopback, 認証なし / 設問19）---
 neighbor 10.1.254.5 remote-as 65001
 neighbor 10.1.254.5 update-source Loopback0
 neighbor 10.1.254.5 next-hop-self
 neighbor 10.1.254.3 remote-as 65001
 neighbor 10.1.254.3 update-source Loopback0
 neighbor 10.1.254.3 next-hop-self
 neighbor 10.1.254.4 remote-as 65001
 neighbor 10.1.254.4 update-source Loopback0
 neighbor 10.1.254.4 next-hop-self
 neighbor 10.1.254.1 remote-as 65001
 neighbor 10.1.254.1 update-source Loopback0
 neighbor 10.1.254.1 next-hop-self
 neighbor 10.1.254.2 remote-as 65001
 neighbor 10.1.254.2 update-source Loopback0
 neighbor 10.1.254.2 next-hop-self
 !
 ! --- 集約ルート 10.1.0.0/16 をIP-VPNへ広告（方式1 / 設問20）---
 network 10.1.0.0 mask 255.255.0.0
!
end
